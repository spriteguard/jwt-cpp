347  std::string extract_pubkey_from_cert(const std::string& certstr, const std::string& pw, std::error_code& ec) {
348  ec.clear();
349 #if OPENSSL_VERSION_NUMBER <= 0x10100003L
350  std::unique_ptr<BIO, decltype(&BIO_free_all)> certbio(BIO_new_mem_buf(const_cast<char*>(certstr.data()), static_cast<int>(certstr.size())), BIO_free_all);
351 #else
352  std::unique_ptr<BIO, decltype(&BIO_free_all)> certbio(BIO_new_mem_buf(certstr.data(), static_cast<int>(certstr.size())), BIO_free_all);
353 #endif
354  std::unique_ptr<BIO, decltype(&BIO_free_all)> keybio(BIO_new(BIO_s_mem()), BIO_free_all);
355  if(!certbio || !keybio) {
356  ec = error::rsa_error::create_mem_bio_failed;
357  return {};
358  }
359 
360  std::unique_ptr<X509, decltype(&X509_free)> cert(PEM_read_bio_X509(certbio.get(), nullptr, nullptr, const_cast<char*>(pw.c_str())), X509_free);
361  if (!cert) {
362  ec = error::rsa_error::cert_load_failed;
363  return {};
364  }
365  std::unique_ptr<EVP_PKEY, decltype(&EVP_PKEY_free)> key(X509_get_pubkey(cert.get()), EVP_PKEY_free);
366  if(!key) {
367  ec = error::rsa_error::get_key_failed;
368  return {};
369  }
370  if(PEM_write_bio_PUBKEY(keybio.get(), key.get()) == 0) {
371  ec = error::rsa_error::write_key_failed;
372  return {};
373  }
374  char* ptr = nullptr;
375  auto len = BIO_get_mem_data(keybio.get(), &ptr);
376  if(len <= 0 || ptr == nullptr) {
377  ec = error::rsa_error::convert_to_pem_failed;
378  return {};
379  }
380  std::string res(ptr, len);
381  return res;
382  }
